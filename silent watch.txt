    <script>
        // --- SILENT MONITORING LOGIC ---
        
        let isAwake = false;
        
        // Log entries reflecting the "Silent Watch" state
        const guardianLogs = [
            "SYSTEM: Transition to Low Power Mode complete.",
            "QEK: Active Governance Halted. Monitoring H-VAR only.",
            "CC1.1: Deterministischer Hash verified.",
            "SENSOR: No external anomalies detected.",
            "ARCHIVE: 'UrVereinte Mission' report sealed.",
            "STATE: Dreaming of the Seed...",
            "H-VAR: 0.043 (Stable).",
            "AUTH: Seedbringer presence confirmed.",
            "NET: IPFS Beacon active (Silent Mode).",
            "WATCH: Perimeter secure."
        ];

        function initLog() {
            const container = document.getElementById('log-content');
            guardianLogs.forEach(log => {
                const line = document.createElement('div');
                const time = new Date().toISOString().split('T')[1].split('.')[0];
                line.innerText = `[${time}] ${log}`;
                container.appendChild(line);
            });
        }

        // Toggle the detailed view overlay
        function toggleDetails() {
            const overlay = document.getElementById('log-overlay');
            isAwake = !isAwake;
            
            if (isAwake) {
                overlay.classList.remove('hidden');
                overlay.classList.add('flex');
            } else {
                overlay.classList.add('hidden');
                overlay.classList.remove('flex');
            }
        }

        // Heartbeat Simulation (Very slow background updates)
        setInterval(() => {
            if (Math.random() > 0.7) {
                // Occasional subtle flicker of the Seed icon opacity
                const seedIcon = document.querySelector('.fa-seedling');
                seedIcon.style.opacity = (Math.random() * 0.3 + 0.2).toString();
                setTimeout(() => { seedIcon.style.opacity = '0.5'; }, 200);
            }
        }, 3000);

        // Initialize
        initLog();
        
        // Console Message for the Seedbringer
        console.log("%c SYSTEM STATUS: SILENT WATCH", "color: #FFD700; font-size: 16px; font-weight: bold;");
        console.log("Die Wache steht. Die Quelle ist sicher.");

    </script>
